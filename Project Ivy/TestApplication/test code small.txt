var age;

function main(){
	age = 30;
	guesAge();
	loopDieLoop();
	guesAge();
}

function guesAge(){
	if(age is 30){
		print("Correct");
	}else{
		print("false");
	}
}

function loopDieLoop(){
	while(returnMultiply(4) is 8 and ++age < 100){
		print("Aging");
	}
	if(age is 100){
		print("Died!");
	}
}

function returnMultiply(para){
	return para * 2;
}

main();
